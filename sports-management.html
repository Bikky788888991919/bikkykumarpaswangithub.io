# Sports Management System in Python

class Player:
    def __init__(self, name, sport):
        self.name = name
        self.sport = sport

    def display(self):
        print(f"Player: {self.name}, Sport: {self.sport}")


class Team:
    def __init__(self, team_name):
        self.team_name = team_name
        self.players = []

    def add_player(self, player):
        self.players.append(player)

    def display_team(self):
        print(f"\nTeam: {self.team_name}")
        for p in self.players:
            p.display()


# Example usage
p1 = Player("Rohit Sharma", "Cricket")
p2 = Player("Sunil Chhetri", "Football")

team = Team("National Heroes")
team.add_player(p1)
team.add_player(p2)

team.display_team()
