import csv
import pickle

CSV_FILE = "sports_data.csv"
BIN_FILE = "sports_data.dat"

# ---------- Utility Functions ----------

def load_data():
    """Load existing data from CSV file (if exists)."""
    data = []
    try:
        with open(CSV_FILE, "r", newline='') as f:
            reader = csv.reader(f)
            data = list(reader)
    except FileNotFoundError:
        pass  # file doesnâ€™t exist yet
    return data

def save_data_csv(data):
    """Save data to CSV file."""
    with open(CSV_FILE, "w", newline='') as f:
        writer = csv.writer(f)
        writer.writerows(data)

def save_data_bin(data):
    """Save data to binary file."""
    with open(BIN_FILE, "wb") as f:
        pickle.dump(data, f)

def read_data_bin():
    """Read data from binary file safely (handle EOFError)."""
    try:
        with open(BIN_FILE, "rb") as f:
            try:
                return pickle.load(f)
            except EOFError:
                return []  # file is empty
    except FileNotFoundError:
        return []  # file doesnâ€™t exist

# ---------- Core Functions ----------

def append_data():
    name = input("Enter player name: ")
    sport = input("Enter sport: ")
    age = input("Enter age: ")

    data = load_data()
    data.append([name, sport, age])

    save_data_csv(data)
    save_data_bin(data)
    print("âœ… Data added successfully to CSV and Binary files!")

def search_data():
    keyword = input("Enter player name to search: ").lower()

    # Search in CSV data
    csv_data = load_data()
    found = [row for row in csv_data if keyword in row[0].lower()]

    if found:
        print("\nğŸ” Found in CSV:")
        for row in found:
            print(row)
    else:
        print("\nâŒ Not found in CSV file.")

    # Search in Binary data
    bin_data = read_data_bin()
    found_bin = [row for row in bin_data if keyword in row[0].lower()]

    if found_bin:
        print("\nğŸ” Found in Binary file:")
        for row in found_bin:
            print(row)
    else:
        print("\nâŒ Not found in Binary file.")

def display_all():
    """Display all records from CSV file"""
    data = load_data()
    if not data:
        print("\nâš  No records available.")
    else:
        print("\nğŸ“‹ All Players Records:")
        for row in data:
            print(row)

# Sports Management System in Python

class Player:
    def __init__(self, name, sport):
        self.name = name
        self.sport = sport

    def display(self):
        print(f"Player: {self.name}, Sport: {self.sport}")


class Team:
    def __init__(self, team_name):
        self.team_name = team_name
        self.players = []

    def add_player(self, player):
        self.players.append(player)

    def display_team(self):
        print(f"\nTeam: {self.team_name}")
        for p in self.players:
            p.display()


# Example usage
p1 = Player("Rohit Sharma", "Cricket")
p2 = Player("Sunil Chhetri", "Football")

team = Team("National Heroes")
team.add_player(p1)
team.add_player(p2)

team.display_team()
